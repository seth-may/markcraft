\begin{abstract}
We prove that the sequence defined by $a_{n+1} = \frac{1}{2}(a_n + \frac{k}{a_n})$
converges to $\sqrt{k}$ for any $k > 0$ and $a_0 > 0$.
\end{abstract}

\section{Introduction}
Let $k > 0$ be a positive real number. The \emph{Babylonian method}
defines a sequence $\{a_n\}_{n \in \N}$ by:
\begin{equation}\label{eq:recurrence}
    a_{n+1} = \frac{1}{2}\left(a_n + \frac{k}{a_n}\right), \quad a_0 > 0
\end{equation}

\begin{theorem}[Convergence]\label{thm:main}
For any $a_0 > 0$ and $k > 0$, the sequence~\eqref{eq:recurrence}
satisfies $\lim_{n \to \infty} a_n = \sqrt{k}$.
\end{theorem}

\begin{proof}
By the AM-GM inequality, for $n \geq 1$:
\[
    a_{n+1} = \frac{a_n^2 + k}{2a_n} \geq \frac{2\sqrt{a_n^2 \cdot k}}{2a_n} = \sqrt{k}
\]
Thus $a_n \geq \sqrt{k}$ for all $n \geq 1$. Now consider:
\[
    a_{n+1} - \sqrt{k} = \frac{(a_n - \sqrt{k})^2}{2a_n} \geq 0
\]
The sequence is monotonically decreasing and bounded below.
By the \textbf{Monotone Convergence Theorem}, $\lim a_n = L$ exists.
Taking limits: $L = \frac{1}{2}(L + k/L) \Rightarrow L = \sqrt{k}$.
\end{proof}
